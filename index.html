# 1. 创建项目
		moon new my_agri_project
		cd my_agri_project

		# 2. 添加 AgriDroneSwarm 库
		moon add liushiman1222/AgriDroneSwarm

		# 3. 在 main.mbt 中使用
		import { Vector3D, PID, Point, a_star } from "liushiman1222/AgriDroneSwarm"

		fn main {
			let v = Vector3D::new(1.0, 2.0, 3.0)
			println("向量长度: \{v.length()}")
		}

		# 4. 运行
		moon run .
